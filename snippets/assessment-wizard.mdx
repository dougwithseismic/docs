import React, { useState } from 'react';

export const AssessmentWizard = () => {
  const [currentStep, setCurrentStep] = useState(0);
  const [assessmentData, setAssessmentData] = useState({
    // Service Information
    serviceName: '',
    currentRevenue: '',
    timeInvestment: '',
    teamMembers: '',
    serviceDescription: '',
    clientOutcomes: '',
    pricingModel: '',

    // Scoring Dimensions
    scores: {
      marketDemand: {
        frequency: 0,
        marketSize: 0,
        urgency: 0,
        average: 0
      },
      differentiation: {
        uniqueness: 0,
        competition: 0,
        ipProtection: 0,
        average: 0
      },
      scalability: {
        automation: 0,
        standardization: 0,
        resources: 0,
        average: 0
      },
      technicalFeasibility: {
        complexity: 0,
        teamCapabilities: 0,
        timeToMarket: 0,
        average: 0
      },
      revenuePotential: {
        pricingPower: 0,
        revenueModel: 0,
        expansion: 0,
        average: 0
      }
    },

    // Strategic Fit
    strategicFit: {
      visionAlignment: 0,
      teamExpertise: 0,
      clientImpact: 0,
      brandFit: 0,
      average: 0
    },

    // Results
    compositeScore: 0,
    priorityQuadrant: '',
    recommendedAction: ''
  });

  const steps = [
    'Service Inventory',
    'Market Demand',
    'Differentiation',
    'Scalability',
    'Technical Feasibility',
    'Revenue Potential',
    'Strategic Fit',
    'Results & Priority'
  ];

  const calculateAverage = (scores) => {
    const values = Object.values(scores).filter(v => typeof v === 'number' && v !== 0);
    return values.length > 0 ? (values.reduce((a, b) => a + b, 0) / values.length).toFixed(2) : 0;
  };

  const calculateCompositeScore = () => {
    const { scores } = assessmentData;
    const weighted =
      (scores.marketDemand.average * 0.25) +
      (scores.differentiation.average * 0.20) +
      (scores.scalability.average * 0.25) +
      (scores.technicalFeasibility.average * 0.15) +
      (scores.revenuePotential.average * 0.15);
    return weighted.toFixed(2);
  };

  const determinePriorityQuadrant = (compositeScore, strategicFitScore) => {
    const composite = parseFloat(compositeScore);
    const strategic = parseFloat(strategicFitScore);

    if (composite >= 3.5 && strategic >= 3.5) return 'Priority 1: Build Now 游릭';
    if (composite >= 3.5 && strategic < 3.5) return 'Priority 2: Evaluate 游리';
    if (composite < 3.5 && strategic >= 3.5) return 'Priority 3: Monitor 游댯';
    return 'Priority 4: Avoid 游댮';
  };

  const getRecommendedAction = (quadrant) => {
    const actions = {
      'Priority 1: Build Now 游릭': 'Immediate development with dedicated resources. Strong business case with perfect strategic alignment.',
      'Priority 2: Evaluate 游리': 'Strong business potential but requires strategic alignment. Consider partnerships or pivots.',
      'Priority 3: Monitor 游댯': 'Good strategic fit but weak business case. Monitor market changes for future consideration.',
      'Priority 4: Avoid 游댮': 'Poor business case and strategic fit. Deprioritize and focus resources elsewhere.'
    };
    return actions[quadrant] || '';
  };

  const handleNext = () => {
    if (currentStep === steps.length - 1) {
      // Calculate final results
      const compositeScore = calculateCompositeScore();
      const quadrant = determinePriorityQuadrant(compositeScore, assessmentData.strategicFit.average);
      setAssessmentData(prev => ({
        ...prev,
        compositeScore,
        priorityQuadrant: quadrant,
        recommendedAction: getRecommendedAction(quadrant)
      }));
    } else {
      setCurrentStep(prev => Math.min(prev + 1, steps.length - 1));
    }
  };

  const handlePrevious = () => {
    setCurrentStep(prev => Math.max(prev - 1, 0));
  };

  const updateScore = (category, subcategory, value) => {
    setAssessmentData(prev => {
      const newData = { ...prev };
      if (category === 'strategicFit') {
        newData.strategicFit[subcategory] = value;
        newData.strategicFit.average = calculateAverage({
          visionAlignment: newData.strategicFit.visionAlignment,
          teamExpertise: newData.strategicFit.teamExpertise,
          clientImpact: newData.strategicFit.clientImpact,
          brandFit: newData.strategicFit.brandFit
        });
      } else {
        newData.scores[category][subcategory] = value;
        newData.scores[category].average = calculateAverage({
          ...newData.scores[category],
          average: undefined
        });
      }
      return newData;
    });
  };

  const ScoreInput = ({ label, value, onChange, description }) => (
    <div style={{ marginBottom: '20px' }}>
      <label style={{ display: 'block', marginBottom: '8px', fontWeight: 'bold' }}>
        {label}
      </label>
      {description && (
        <p style={{ fontSize: '14px', color: '#6b7280', marginBottom: '8px' }}>{description}</p>
      )}
      <div style={{ display: 'flex', gap: '10px', alignItems: 'center' }}>
        {[1, 2, 3, 4, 5].map(score => (
          <button
            key={score}
            onClick={() => onChange(score)}
            style={{
              width: '40px',
              height: '40px',
              borderRadius: '8px',
              border: value === score ? '2px solid #16A34A' : '1px solid #d1d5db',
              backgroundColor: value === score ? '#dcfce7' : 'white',
              cursor: 'pointer',
              fontWeight: value === score ? 'bold' : 'normal',
              color: value === score ? '#16A34A' : '#374151'
            }}
          >
            {score}
          </button>
        ))}
        <span style={{ marginLeft: '10px', fontWeight: 'bold', color: '#16A34A' }}>
          {value > 0 ? `Score: ${value}` : ''}
        </span>
      </div>
    </div>
  );

  const renderStepContent = () => {
    switch (currentStep) {
      case 0: // Service Inventory
        return (
          <div>
            <h3 style={{ marginBottom: '24px' }}>Service Information</h3>
            <div style={{ display: 'grid', gap: '16px' }}>
              <div>
                <label style={{ display: 'block', marginBottom: '8px', fontWeight: 'bold' }}>
                  Service Name
                </label>
                <input
                  type="text"
                  value={assessmentData.serviceName}
                  onChange={(e) => setAssessmentData(prev => ({ ...prev, serviceName: e.target.value }))}
                  style={{
                    width: '100%',
                    padding: '8px 12px',
                    border: '1px solid #d1d5db',
                    borderRadius: '6px'
                  }}
                  placeholder="Enter service name"
                />
              </div>

              <div>
                <label style={{ display: 'block', marginBottom: '8px', fontWeight: 'bold' }}>
                  Current Annual Revenue
                </label>
                <input
                  type="text"
                  value={assessmentData.currentRevenue}
                  onChange={(e) => setAssessmentData(prev => ({ ...prev, currentRevenue: e.target.value }))}
                  style={{
                    width: '100%',
                    padding: '8px 12px',
                    border: '1px solid #d1d5db',
                    borderRadius: '6px'
                  }}
                  placeholder="$0"
                />
              </div>

              <div>
                <label style={{ display: 'block', marginBottom: '8px', fontWeight: 'bold' }}>
                  Time Investment (hours per delivery)
                </label>
                <input
                  type="text"
                  value={assessmentData.timeInvestment}
                  onChange={(e) => setAssessmentData(prev => ({ ...prev, timeInvestment: e.target.value }))}
                  style={{
                    width: '100%',
                    padding: '8px 12px',
                    border: '1px solid #d1d5db',
                    borderRadius: '6px'
                  }}
                  placeholder="0 hours"
                />
              </div>

              <div>
                <label style={{ display: 'block', marginBottom: '8px', fontWeight: 'bold' }}>
                  Team Members Required
                </label>
                <input
                  type="text"
                  value={assessmentData.teamMembers}
                  onChange={(e) => setAssessmentData(prev => ({ ...prev, teamMembers: e.target.value }))}
                  style={{
                    width: '100%',
                    padding: '8px 12px',
                    border: '1px solid #d1d5db',
                    borderRadius: '6px'
                  }}
                  placeholder="Number of team members"
                />
              </div>

              <div>
                <label style={{ display: 'block', marginBottom: '8px', fontWeight: 'bold' }}>
                  Service Description
                </label>
                <textarea
                  value={assessmentData.serviceDescription}
                  onChange={(e) => setAssessmentData(prev => ({ ...prev, serviceDescription: e.target.value }))}
                  style={{
                    width: '100%',
                    padding: '8px 12px',
                    border: '1px solid #d1d5db',
                    borderRadius: '6px',
                    minHeight: '100px'
                  }}
                  placeholder="Describe the service and its key processes"
                />
              </div>

              <div>
                <label style={{ display: 'block', marginBottom: '8px', fontWeight: 'bold' }}>
                  Client Outcomes
                </label>
                <textarea
                  value={assessmentData.clientOutcomes}
                  onChange={(e) => setAssessmentData(prev => ({ ...prev, clientOutcomes: e.target.value }))}
                  style={{
                    width: '100%',
                    padding: '8px 12px',
                    border: '1px solid #d1d5db',
                    borderRadius: '6px',
                    minHeight: '100px'
                  }}
                  placeholder="What outcomes does this service deliver for clients?"
                />
              </div>
            </div>
          </div>
        );

      case 1: // Market Demand
        return (
          <div>
            <h3 style={{ marginBottom: '24px' }}>游늳 Market Demand Assessment</h3>
            <ScoreInput
              label="Request Frequency"
              value={assessmentData.scores.marketDemand.frequency}
              onChange={(value) => updateScore('marketDemand', 'frequency', value)}
              description="1: <2 requests/year | 2: 2-4 | 3: 5-10 | 4: 11-20 | 5: 20+"
            />
            <ScoreInput
              label="Market Size"
              value={assessmentData.scores.marketDemand.marketSize}
              onChange={(value) => updateScore('marketDemand', 'marketSize', value)}
              description="1: <100 customers | 2: 100-500 | 3: 500-2K | 4: 2K-10K | 5: 10K+"
            />
            <ScoreInput
              label="Urgency Level"
              value={assessmentData.scores.marketDemand.urgency}
              onChange={(value) => updateScore('marketDemand', 'urgency', value)}
              description="1: Nice to have | 2: Can delay | 3: Important | 4: High priority | 5: Critical"
            />
            <div style={{ marginTop: '24px', padding: '16px', backgroundColor: '#f3f4f6', borderRadius: '8px' }}>
              <strong>Average Score: {assessmentData.scores.marketDemand.average}</strong>
            </div>
          </div>
        );

      case 2: // Differentiation
        return (
          <div>
            <h3 style={{ marginBottom: '24px' }}>游꿢 Differentiation Assessment</h3>
            <ScoreInput
              label="Uniqueness"
              value={assessmentData.scores.differentiation.uniqueness}
              onChange={(value) => updateScore('differentiation', 'uniqueness', value)}
              description="1: Standard | 2: Minor variations | 3: Some proprietary | 4: Significantly different | 5: Completely unique"
            />
            <ScoreInput
              label="Competition Level"
              value={assessmentData.scores.differentiation.competition}
              onChange={(value) => updateScore('differentiation', 'competition', value)}
              description="1: Many competitors | 2: Some | 3: Few | 4: Limited | 5: No competitors"
            />
            <ScoreInput
              label="IP Protection Potential"
              value={assessmentData.scores.differentiation.ipProtection}
              onChange={(value) => updateScore('differentiation', 'ipProtection', value)}
              description="1: Cannot protect | 2: Limited | 3: Some elements | 4: Significant | 5: Strong protection"
            />
            <div style={{ marginTop: '24px', padding: '16px', backgroundColor: '#f3f4f6', borderRadius: '8px' }}>
              <strong>Average Score: {assessmentData.scores.differentiation.average}</strong>
            </div>
          </div>
        );

      case 3: // Scalability
        return (
          <div>
            <h3 style={{ marginBottom: '24px' }}>游 Scalability Assessment</h3>
            <ScoreInput
              label="Automation Potential"
              value={assessmentData.scores.scalability.automation}
              onChange={(value) => updateScore('scalability', 'automation', value)}
              description="1: Manual | 2: Some automation | 3: Significant | 4: Mostly | 5: Fully automatable"
            />
            <ScoreInput
              label="Standardization Level"
              value={assessmentData.scores.scalability.standardization}
              onChange={(value) => updateScore('scalability', 'standardization', value)}
              description="1: Highly customized | 2: Some standards | 3: Moderate | 4: Highly standardized | 5: Completely standard"
            />
            <ScoreInput
              label="Resource Requirements"
              value={assessmentData.scores.scalability.resources}
              onChange={(value) => updateScore('scalability', 'resources', value)}
              description="1: High needs | 2: Moderate | 3: Some | 4: Minimal | 5: No ongoing needs"
            />
            <div style={{ marginTop: '24px', padding: '16px', backgroundColor: '#f3f4f6', borderRadius: '8px' }}>
              <strong>Average Score: {assessmentData.scores.scalability.average}</strong>
            </div>
          </div>
        );

      case 4: // Technical Feasibility
        return (
          <div>
            <h3 style={{ marginBottom: '24px' }}>丘뙖잺 Technical Feasibility Assessment</h3>
            <ScoreInput
              label="Implementation Complexity"
              value={assessmentData.scores.technicalFeasibility.complexity}
              onChange={(value) => updateScore('technicalFeasibility', 'complexity', value)}
              description="1: Extremely complex | 2: Complex | 3: Moderate | 4: Simple | 5: Very simple"
            />
            <ScoreInput
              label="Team Capabilities"
              value={assessmentData.scores.technicalFeasibility.teamCapabilities}
              onChange={(value) => updateScore('technicalFeasibility', 'teamCapabilities', value)}
              description="1: No capabilities | 2: Limited | 3: Some | 4: Strong team | 5: Excellent team"
            />
            <ScoreInput
              label="Time to Market"
              value={assessmentData.scores.technicalFeasibility.timeToMarket}
              onChange={(value) => updateScore('technicalFeasibility', 'timeToMarket', value)}
              description="1: >12 months | 2: 6-12 months | 3: 3-6 months | 4: 1-3 months | 5: <1 month"
            />
            <div style={{ marginTop: '24px', padding: '16px', backgroundColor: '#f3f4f6', borderRadius: '8px' }}>
              <strong>Average Score: {assessmentData.scores.technicalFeasibility.average}</strong>
            </div>
          </div>
        );

      case 5: // Revenue Potential
        return (
          <div>
            <h3 style={{ marginBottom: '24px' }}>游눯 Revenue Potential Assessment</h3>
            <ScoreInput
              label="Pricing Power"
              value={assessmentData.scores.revenuePotential.pricingPower}
              onChange={(value) => updateScore('revenuePotential', 'pricingPower', value)}
              description="1: Low pricing | 2: Below average | 3: Market rate | 4: Premium | 5: Exceptional"
            />
            <ScoreInput
              label="Revenue Model"
              value={assessmentData.scores.revenuePotential.revenueModel}
              onChange={(value) => updateScore('revenuePotential', 'revenueModel', value)}
              description="1: One-time | 2: Limited recurring | 3: Some recurring | 4: Strong recurring | 5: SaaS/Subscription"
            />
            <ScoreInput
              label="Expansion Opportunity"
              value={assessmentData.scores.revenuePotential.expansion}
              onChange={(value) => updateScore('revenuePotential', 'expansion', value)}
              description="1: Current clients only | 2: Small | 3: Moderate | 4: Significant | 5: Massive opportunity"
            />
            <div style={{ marginTop: '24px', padding: '16px', backgroundColor: '#f3f4f6', borderRadius: '8px' }}>
              <strong>Average Score: {assessmentData.scores.revenuePotential.average}</strong>
            </div>
          </div>
        );

      case 6: // Strategic Fit
        return (
          <div>
            <h3 style={{ marginBottom: '24px' }}>游꿢 Strategic Fit Assessment</h3>
            <ScoreInput
              label="Vision Alignment"
              value={assessmentData.strategicFit.visionAlignment}
              onChange={(value) => updateScore('strategicFit', 'visionAlignment', value)}
              description="How well does this align with your agency's vision?"
            />
            <ScoreInput
              label="Team Expertise"
              value={assessmentData.strategicFit.teamExpertise}
              onChange={(value) => updateScore('strategicFit', 'teamExpertise', value)}
              description="How well does your team's expertise match this opportunity?"
            />
            <ScoreInput
              label="Client Impact"
              value={assessmentData.strategicFit.clientImpact}
              onChange={(value) => updateScore('strategicFit', 'clientImpact', value)}
              description="How significantly will this impact your target clients?"
            />
            <ScoreInput
              label="Brand Fit"
              value={assessmentData.strategicFit.brandFit}
              onChange={(value) => updateScore('strategicFit', 'brandFit', value)}
              description="How well does this fit with your brand positioning?"
            />
            <div style={{ marginTop: '24px', padding: '16px', backgroundColor: '#f3f4f6', borderRadius: '8px' }}>
              <strong>Average Score: {assessmentData.strategicFit.average}</strong>
            </div>
          </div>
        );

      case 7: // Results
        const finalCompositeScore = calculateCompositeScore();
        const finalQuadrant = determinePriorityQuadrant(finalCompositeScore, assessmentData.strategicFit.average);
        const finalAction = getRecommendedAction(finalQuadrant);

        return (
          <div>
            <h3 style={{ marginBottom: '24px' }}>游늵 Assessment Results</h3>

            <div style={{
              padding: '24px',
              backgroundColor: '#f9fafb',
              borderRadius: '12px',
              marginBottom: '24px'
            }}>
              <h4 style={{ marginBottom: '16px' }}>Service: {assessmentData.serviceName || 'Unnamed Service'}</h4>

              <div style={{ display: 'grid', gap: '16px' }}>
                <div>
                  <strong>Composite Score:</strong>
                  <div style={{
                    fontSize: '36px',
                    color: '#16A34A',
                    fontWeight: 'bold',
                    marginTop: '8px'
                  }}>
                    {finalCompositeScore}/5.00
                  </div>
                </div>

                <div>
                  <strong>Strategic Fit Score:</strong>
                  <div style={{
                    fontSize: '24px',
                    color: '#3b82f6',
                    marginTop: '8px'
                  }}>
                    {assessmentData.strategicFit.average}/5.00
                  </div>
                </div>

                <div style={{
                  padding: '16px',
                  borderRadius: '8px',
                  backgroundColor: finalQuadrant.includes('Priority 1') ? '#dcfce7' :
                                 finalQuadrant.includes('Priority 2') ? '#fef3c7' :
                                 finalQuadrant.includes('Priority 3') ? '#dbeafe' : '#fee2e2'
                }}>
                  <strong>Priority Classification:</strong>
                  <div style={{ fontSize: '20px', marginTop: '8px' }}>
                    {finalQuadrant}
                  </div>
                </div>

                <div>
                  <strong>Recommended Action:</strong>
                  <p style={{ marginTop: '8px', lineHeight: '1.6' }}>
                    {finalAction}
                  </p>
                </div>
              </div>
            </div>

            <div style={{ marginBottom: '24px' }}>
              <h4 style={{ marginBottom: '16px' }}>Scoring Breakdown</h4>
              <table style={{ width: '100%', borderCollapse: 'collapse' }}>
                <thead>
                  <tr style={{ backgroundColor: '#f3f4f6' }}>
                    <th style={{ padding: '12px', textAlign: 'left', borderBottom: '2px solid #d1d5db' }}>
                      Dimension
                    </th>
                    <th style={{ padding: '12px', textAlign: 'center', borderBottom: '2px solid #d1d5db' }}>
                      Weight
                    </th>
                    <th style={{ padding: '12px', textAlign: 'center', borderBottom: '2px solid #d1d5db' }}>
                      Score
                    </th>
                    <th style={{ padding: '12px', textAlign: 'center', borderBottom: '2px solid #d1d5db' }}>
                      Weighted
                    </th>
                  </tr>
                </thead>
                <tbody>
                  <tr>
                    <td style={{ padding: '12px', borderBottom: '1px solid #e5e7eb' }}>
                      游늳 Market Demand
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center', borderBottom: '1px solid #e5e7eb' }}>
                      25%
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center', borderBottom: '1px solid #e5e7eb' }}>
                      {assessmentData.scores.marketDemand.average}
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center', borderBottom: '1px solid #e5e7eb' }}>
                      {(assessmentData.scores.marketDemand.average * 0.25).toFixed(2)}
                    </td>
                  </tr>
                  <tr>
                    <td style={{ padding: '12px', borderBottom: '1px solid #e5e7eb' }}>
                      游꿢 Differentiation
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center', borderBottom: '1px solid #e5e7eb' }}>
                      20%
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center', borderBottom: '1px solid #e5e7eb' }}>
                      {assessmentData.scores.differentiation.average}
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center', borderBottom: '1px solid #e5e7eb' }}>
                      {(assessmentData.scores.differentiation.average * 0.20).toFixed(2)}
                    </td>
                  </tr>
                  <tr>
                    <td style={{ padding: '12px', borderBottom: '1px solid #e5e7eb' }}>
                      游 Scalability
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center', borderBottom: '1px solid #e5e7eb' }}>
                      25%
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center', borderBottom: '1px solid #e5e7eb' }}>
                      {assessmentData.scores.scalability.average}
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center', borderBottom: '1px solid #e5e7eb' }}>
                      {(assessmentData.scores.scalability.average * 0.25).toFixed(2)}
                    </td>
                  </tr>
                  <tr>
                    <td style={{ padding: '12px', borderBottom: '1px solid #e5e7eb' }}>
                      丘뙖잺 Technical Feasibility
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center', borderBottom: '1px solid #e5e7eb' }}>
                      15%
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center', borderBottom: '1px solid #e5e7eb' }}>
                      {assessmentData.scores.technicalFeasibility.average}
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center', borderBottom: '1px solid #e5e7eb' }}>
                      {(assessmentData.scores.technicalFeasibility.average * 0.15).toFixed(2)}
                    </td>
                  </tr>
                  <tr>
                    <td style={{ padding: '12px', borderBottom: '1px solid #e5e7eb' }}>
                      游눯 Revenue Potential
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center', borderBottom: '1px solid #e5e7eb' }}>
                      15%
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center', borderBottom: '1px solid #e5e7eb' }}>
                      {assessmentData.scores.revenuePotential.average}
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center', borderBottom: '1px solid #e5e7eb' }}>
                      {(assessmentData.scores.revenuePotential.average * 0.15).toFixed(2)}
                    </td>
                  </tr>
                  <tr style={{ fontWeight: 'bold', backgroundColor: '#f3f4f6' }}>
                    <td style={{ padding: '12px' }}>
                      Total
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center' }}>
                      100%
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center' }}>
                      -
                    </td>
                    <td style={{ padding: '12px', textAlign: 'center' }}>
                      {finalCompositeScore}
                    </td>
                  </tr>
                </tbody>
              </table>
            </div>

            <div style={{
              padding: '16px',
              backgroundColor: '#e0f2fe',
              borderRadius: '8px',
              marginTop: '24px'
            }}>
              <strong>游눠 Next Steps:</strong>
              <ul style={{ marginTop: '8px', paddingLeft: '20px' }}>
                {finalQuadrant.includes('Priority 1') && (
                  <>
                    <li>Begin immediate validation with target customers</li>
                    <li>Allocate dedicated resources for development</li>
                    <li>Create detailed project plan with aggressive timelines</li>
                    <li>Start MVP development within 2 weeks</li>

                {finalQuadrant.includes('Priority 2') && (
                  <>
                    <li>Conduct competitive analysis</li>
                    <li>Validate key assumptions with market research</li>

                {finalQuadrant.includes('Priority 3') && (
                  <>
                    <li>Monitor market trends quarterly</li>
                    <li>Gather ongoing customer feedback</li>
                    <li>Work on improving weak scoring areas</li>
                    <li>Revisit assessment in 6 months</li>
                  </>
                )}
                {finalQuadrant.includes('Priority 4') && (
                  <>
                    <li>Document findings for future reference</li>
                    <li>Focus resources on higher-scoring opportunities</li>
                    <li>Review annually for market changes</li>
                    <li>Consider retiring if currently offered</li>
                  </>
                )}
              </ul>
            </div>
          </div>
        );

      default:
        return null;
    }
  };

  return (
    <div style={{
      maxWidth: '800px',
      margin: '0 auto',
      padding: '24px',
      backgroundColor: 'white',
      borderRadius: '12px',
      boxShadow: '0 1px 3px 0 rgba(0, 0, 0, 0.1), 0 1px 2px 0 rgba(0, 0, 0, 0.06)'
    }}>
      {/* Progress Bar */}
      <div style={{ marginBottom: '32px' }}>
        <div style={{ display: 'flex', justifyContent: 'space-between', marginBottom: '8px' }}>
          <span style={{ fontSize: '14px', color: '#6b7280' }}>
            Step {currentStep + 1} of {steps.length}
          </span>
          <span style={{ fontSize: '14px', fontWeight: 'bold' }}>
            {steps[currentStep]}
          </span>
        </div>
        <div style={{
          height: '8px',
          backgroundColor: '#e5e7eb',
          borderRadius: '4px',
          overflow: 'hidden'
        }}>
          <div style={{
            height: '100%',
            backgroundColor: '#16A34A',
            borderRadius: '4px',
            width: `${((currentStep + 1) / steps.length) * 100}%`,
            transition: 'width 0.3s ease'
          }} />
        </div>
      </div>

      {/* Step Content */}
      <div style={{ minHeight: '400px' }}>
        {renderStepContent()}
      </div>

      {/* Navigation Buttons */}
      <div style={{
        display: 'flex',
        justifyContent: 'space-between',
        marginTop: '32px',
        paddingTop: '24px',
        borderTop: '1px solid #e5e7eb'
      }}>
        <button
          onClick={handlePrevious}
          disabled={currentStep === 0}
          style={{
            padding: '12px 24px',
            backgroundColor: currentStep === 0 ? '#e5e7eb' : 'white',
            border: '1px solid #d1d5db',
            borderRadius: '8px',
            cursor: currentStep === 0 ? 'not-allowed' : 'pointer',
            opacity: currentStep === 0 ? 0.5 : 1
          }}
        >
          Previous
        </button>

        <button
          onClick={handleNext}
          style={{
            padding: '12px 24px',
            backgroundColor: '#16A34A',
            color: 'white',
            border: 'none',
            borderRadius: '8px',
            cursor: 'pointer',
            fontWeight: 'bold'
          }}
        >
          {currentStep === steps.length - 1 ? 'View Results' : 'Next'}
        </button>
      </div>
    </div>
  );
};
