<script dangerouslySetInnerHTML={{__html: `
function calculateRecommendation() {
  const size = document.getElementById('company-size').value;
  const industry = document.getElementById('company-industry').value;

  if (!size || !industry) {
    alert('Please fill in all fields');
    return;
  }

  const recommendations = {
    small: {
      title: "Starter Productization Package",
      description: "Perfect for agencies under 10 people looking to automate their first processes"
    },
    medium: {
      title: "Growth Acceleration Package",
      description: "Ideal for scaling agencies ready to 10x their capacity"
    },
    large: {
      title: "Enterprise Transformation",
      description: "Complete digital transformation for established agencies"
    }
  };

  const sizeKey = parseInt(size) < 10 ? 'small' :
                  parseInt(size) < 50 ? 'medium' : 'large';

  const result = recommendations[sizeKey];

  // Track with global tracker
  if (window.WithSeismicTracker) {
    window.WithSeismicTracker.captureCompany({
      size: size,
      industry: industry
    });
    window.WithSeismicTracker.trackCalculation({
      tool: 'company-size-calculator',
      companySize: size,
      industry: industry,
      recommendation: result.title
    });
  }

  // Show result
  const resultDiv = document.getElementById('result');
  resultDiv.innerHTML = \`
    <h4 style="color: #7C3AED; font-weight: 600; margin-bottom: 8px;">
      \${result.title}
    </h4>
    <p style="color: #6b7280;">\${result.description}</p>
  \`;
  resultDiv.style.display = 'block';
}

// Track tool view
if (window.WithSeismicTracker) {
  window.WithSeismicTracker.trackTool('company-size-calculator', { action: 'viewed' });
}
`}} />
