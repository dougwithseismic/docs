```javascript
// Quality score calculation
const calculateQualityScore = (checks) => {
  const weights = {
    critical: 3,
    important: 2,
    nice_to_have: 1,
  };

  let totalScore = 0;
  let maxScore = 0;

  checks.forEach((check) => {
    maxScore += weights[check.priority];
    if (check.passed) {
      totalScore += weights[check.priority];
    }
  });

  return (totalScore / maxScore) * 100;
};
```
